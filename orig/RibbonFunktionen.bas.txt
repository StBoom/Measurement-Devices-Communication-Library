#If VBA7 Then
Function GetRibbon(ByVal lRibbonPointer As LongPtr) As Object
#Else
Function GetRibbon(ByVal lRibbonPointer As Long) As Object
#End If
        Dim objRibbon As Object
        CopyMemory objRibbon, lRibbonPointer, LenB(lRibbonPointer)
        Set GetRibbon = objRibbon
        Set objRibbon = Nothing
End Function

Public Sub Init()
    'Initialisieren der INI Funktion Pfad und Objekt erzeugen
    Path = Environ("APPDATA") & "\Microsoft\AddIns\INSTRUMENT_VISA.ini"
    Set iniClass = New clsIni   ' Instanz erstellen
    If Dir(Path) = "" Then
        generate_standard_ini
    End If
End Sub
Public Function generate_standard_ini()

        iniClass.WritePrivateProfileString Path, "Ribbon", "check_S1", "Falsch"
        iniClass.WritePrivateProfileString Path, "Ribbon", "check_S2", "Falsch"
        iniClass.WritePrivateProfileString Path, "Ribbon", "check_S3", "Falsch"
        iniClass.WritePrivateProfileString Path, "Ribbon", "check_S4", "Falsch"
        iniClass.WritePrivateProfileString Path, "Ribbon", "sfile_format", "AUTO"
        iniClass.WritePrivateProfileString Path, "Ribbon", "sfile_format_index", "0"
        iniClass.WritePrivateProfileString Path, "VISA_INSTRUMENTS", "visa_instruments_addr_up", "Geräteaddresse"
        iniClass.WritePrivateProfileString Path, "VISA_INSTRUMENTS", "visa_instruments_addr_down", "Geräteaddresse"
        iniClass.WritePrivateProfileString Path, "equipment", "Anzahl", "2"
        iniClass.WritePrivateProfileString Path, "equipment", "Item1", "Multimeter"
        iniClass.WritePrivateProfileString Path, "equipment", "Item2", "Oszilloskop"
        iniClass.WritePrivateProfileString Path, "Multimeter", "Anzahl", "2"
        iniClass.WritePrivateProfileString Path, "Multimeter", "Item1", "*IDN?"
        iniClass.WritePrivateProfileString Path, "Multimeter", "Item2", "*RST"
        iniClass.WritePrivateProfileString Path, "Oszilloskop", "Anzahl", "2"
        iniClass.WritePrivateProfileString Path, "Oszilloskop", "Item1", "*IDN?"
        iniClass.WritePrivateProfileString Path, "Oszilloskop", "Item2", "*RST"
End Function
Sub RefreshRibbon()
    If objRibbon Is Nothing Then
        Set objRibbon = GetRibbon(iniClass.GetPrivateProfileString(Path, "Ribbon", "Ribbon_obj_id"))   ' -------- Neu eigefügt
        objRibbon.Invalidate
    Else
        objRibbon.Invalidate
    End If
End Sub

'Callback for button0 onAction
Sub actionScreenshot(control As IRibbonControl)
    On Error GoTo ups
    Dim Config As TypeDefs.Confing
    Dim Device As New getVisaFile
    Dim Bild As TypeDefs.Data
    Init
    Config.Address = iniClass.GetPrivateProfileString(Path, "Ribbon", "edit_address")
    Config.Titel = iniClass.GetPrivateProfileString(Path, "Ribbon", "edit_title")
    Config.EinfuegenInSheet = True
    Device.Init Config
    Bild = Device.Screenshot
    filehandler.saveIt Bild, Bild.EinfuegenInSheet
    Exit Sub
ups:
    MsgBox "Fehler in Funkion Screenshot" & vbCrLf & "Fehlermeldung: " & Err.Description & vbCrLf & "Fehler Source: " & Err.source
    End
End Sub
'Callback for button4 onAction
Sub actionSearchInst(control As IRibbonControl)
    On Error GoTo ups
    Dim Config As TypeDefs.Confing
    Dim Devices As New getVisaDevices
    Devices.SearchVisaDevices
    On Error GoTo ArrayLeer
    If UBound(Devices.getDeviceAddr) >= 0 Then
        Init
        iniClass.WritePrivateProfileString Path, "Ribbon", "edit_address", Devices.ChooseDevice
        Call RefreshRibbon
    End If
    Exit Sub
ArrayLeer:
        MsgBox "kein Gerät gefunden, Überüfen sie die Verbindung, oder nutzen sie das R&S Tool zur Gerätesuche"
    Exit Sub
ups:
    MsgBox "Fehler in Funkion actionSearchInst" & vbCrLf & "Fehlermeldung: " & Err.Description & vbCrLf & "Fehler Source: " & Err.source
    End
End Sub

Sub actionGetSfile(control As IRibbonControl)
    Init
    Dim Config As TypeDefs.Confing

    Dim SFile As TypeDefs.Data
    'Daten Aus Ini Holen (bzw. Ribbon Buttons
    Config.S1 = iniClass.GetPrivateProfileString(Path, "Ribbon", "check_S1")
    Config.S2 = iniClass.GetPrivateProfileString(Path, "Ribbon", "check_S2")
    Config.S3 = iniClass.GetPrivateProfileString(Path, "Ribbon", "check_S3")
    Config.S4 = iniClass.GetPrivateProfileString(Path, "Ribbon", "check_S4")
    Config.Address = iniClass.GetPrivateProfileString(Path, "Ribbon", "edit_address")
    Config.Format = iniClass.GetPrivateProfileString(Path, "Ribbon", "sfile_format")
    'An Funktion Übergeben
    Device.Init Config
    SFile = Device.SFile
    filehandler.saveIt SFile, False
End Sub
