Public Function Screenshot() As TypeDefs.Data
    If initialisert Then
         If InStr(1, mIdn, "Hewlett-Packard, E740", 1) > 0 Then
            Picture_E7402A
        ElseIf InStr(1, mIdn, "Agilent Technologies,E50", 1) > 0 Then
            Picture_E5071C
        ElseIf InStr(1, mIdn, "Rohde&Schwarz,FSW", 1) > 0 Then
            Picture_FSW
        ElseIf InStr(1, mIdn, "AGILENT TECHNOLOGIES,DSO-X", 1) > 0 Or _
        InStr(1, mIdn, "AGILENT TECHNOLOGIES,MSO-X", 1) > 0 Then
            Picture_3034A
        ElseIf InStr(1, mIdn, "Agilent Technologies,N90", 1) > 0 Then
            Picture_PXA_MXA_EXA_CXA
        ElseIf InStr(1, mIdn, "Rohde-Schwarz,ZNB", 1) > 0 Then
            If MsgBox("Soll nur das Aktive Fenster aufgenommen werden", vbYesNo) = vbYes Then
                Picture_ZNB_Act
            Else
                Picture_ZNB_Full
            End If
        ElseIf InStr(1, mIdn, "TEKTRONIX,MSO4", 1) > 0 Or _
        InStr(1, mIdn, "TEKTRONIX,MDO4", 1) > 0 Or _
        InStr(1, mIdn, "TEKTRONIX,MDO3", 1) > 0 Or _
        InStr(1, mIdn, "TEKTRONIX,DPO4", 1) > 0 Then
            Picture_Tek_mdo
        ElseIf InStr(1, mIdn, "TEKTRONIX,TDS30", 1) > 0 Then
            Picture_Tek_tds
        ElseIf InStr(1, mIdn, "AGILENT TECHNOLOGIES,MSO7", 1) > 0 Or _
        InStr(1, mIdn, "AGILENT TECHNOLOGIES,DSO7", 1) > 0 Then
            Picture_InfiniiVision7000B

            MsgBox "Unbekanntes Gerät"
            mFile.Filetype = "NoFile"
        End If

        MsgBox "getVisaFile ist nicht initialisiert worden."
        mFile.Filetype = "NoFile"
    End If

Private Function Sfile_E5071C() As String
    On Error GoTo ups
    Dim Filetyp As Integer
    Dim SX As String
    Filetyp = (mConfig.S1 + mConfig.S2 + mConfig.S3 + mConfig.S4) * -1
    If mConfig.S1 Then
        SX = SX & "1,"
    End If
    If mConfig.S2 Then
        SX = SX & "2,"
    End If
    If mConfig.S3 Then
        SX = SX & "3,"
    End If
    If mConfig.S4 Then
        SX = SX & "4,"
    End If
    If Len(SX) > 1 Then
        SX = Left(SX, Len(SX) - 1)
        mFile.Filetype = "s" & Filetyp & "p"
        mInstrument.WriteString ":SYST:BEEP:COMP:STAT OFF" & vbLf
        mInstrument.WriteString ":MMEM:STOR:SNP:FORM " & mConfig.Format & "; *WAI" & vbLf ' AUTO / DB / MA / RI
        mInstrument.WriteString ":MMEM:STOR:SNP:TYPE:S" & Filetyp & "P " & SX & ";*WAI" & vbLf
        mInstrument.WriteString ":MMEM:STOR:SNP 'D:\SNP01." & mFile.Filetype & "';*WAI" & vbLf
        mInstrument.WriteString ":MMEM:TRAN? 'D:\SNP01." & mFile.Filetype & "';*WAI" & vbLf
        mFile.File = mInstrument.ReadString()
        mInstrument.WriteString ":MMEM:DEL 'D:\SNP01." & mFile.Filetype & "';*WAI" & vbLf
        mFile.File = Right(mFile.File, Len(mFile.File) - 8) ' Datenmüll am Anfang der Datei abschneiden
        mInstrument.IO.Close
    End If
    Exit Function
ups:
    ErrorhandlerVisa mInstrument, "Sfile_E5071C"
End Function
Private Function Sfile_ZNB() As String
    On Error GoTo ups
    Dim Filetyp As Integer
    Dim SX As String
    Dim activeCH As Integer
    Dim Portstatus As String
    Dim ports() As String
    Dim anzahlPorts As Integer
    
    If mConfig.Format = "DB" Then
        mConfig.Format = "LOGP"
    ElseIf mConfig.Format = "AUTO" Or mConfig.Format = "RI" Then
        mConfig.Format = "COMP"
    ElseIf mConfig.Format = "MA" Then
        mConfig.Format = "LINP"
    End If
    Filetyp = (mConfig.S1 + mConfig.S2 + mConfig.S3 + mConfig.S4) * -1
    If mConfig.S1 Then
        SX = SX & "1,"
    End If
    If mConfig.S2 Then
        SX = SX & "2,"
    End If
    If mConfig.S3 Then
        SX = SX & "3,"
    End If
    If mConfig.S4 Then
        SX = SX & "4,"
    End If
    mInstrument.WriteString "INST:NSEL?" & vbLf ' aktiven kanal abfragen
    activeCH = mInstrument.ReadString
    
    mInstrument.WriteString "SOUR:GRO:PPOR?" & vbLf ' aktiven Ports abfragen
    Portstatus = mInstrument.ReadString
    ports = Split(Portstatus, ",")
    anzahlPorts = UBound(ports) - LBound(ports) + 1
    
    If Filetyp > anzahlPorts Then
        MsgBox "Zu viele S Parameter angehakt, für die Anzahl der aktivierten Ports am Gerät"
        End
    End If
    
    If Len(SX) > 1 Then
        SX = Left(SX, Len(SX) - 1)
        mFile.Filetype = "s" & Filetyp & "p"
        mInstrument.WriteString "MMEM:CDIR DEF" & vbLf ' Speicherort angeben
        mInstrument.WriteString "MMEM:STOR:TRAC:PORT " & activeCH & ", " & "'Traces\visatmp." & mFile.Filetype & "', " & mConfig.Format & ", CIMP, " & SX & vbLf ' Trace Speichern"
        'mInstrument.WriteString "MMEM:STOR:TRAC 'Trc1', 'Traces\visatmp." & mFile.Filetype & "' ," & "UNF, " & mConfig.Format & vbLf ' Trace Speichern"
        mInstrument.WriteString "MMEM:DATA? 'Traces\visatmp." & mFile.Filetype & "'" & vbLf ' Trace Ausgeben
        mFile.File = mInstrument.ReadString() 'ausgegebenen Trace Lesen
        mFile.File = Right(mFile.File, Len(mFile.File) - 8) ' Datenmüll am Anfang der Datei abschneiden
        mInstrument.WriteString "MMEM:DEL 'Traces\visatmp." & mFile.Filetype & "'" & vbLf ' Trace Löschen
        mInstrument.IO.Close
    End If
    Exit Function
ups:
    ErrorhandlerVisa mInstrument, "Function Sfile_ZNB"
End Function

Private Function Picture_E5071C()
    On Error GoTo ups
    Dim dispstatus As String
    Dim menustatus As String
    mInstrument.WriteString ":DISP:MENU:STATE 0" & vbLf ' Menü auschalten
    mInstrument.WriteString ":HCOP:SDUM:DATA:FORM PNG" & vbLf ' Format auf PNG stellen
    mInstrument.WriteString ":DISP:IMAG?" & vbLf ' Abfragen wie der Screen des Gerätes eingestellt ist
    dispstatus = mInstrument.ReadString() ' Status sichern
    mInstrument.WriteString ":DISP:IMAG INV" & vbLf ' Screen auf Invertiert stellen
    mInstrument.WriteString ":DISP:SKEY:STAT?" & vbLf ' Abfragen wie das Menü des Gerätes eingestelt ist
    menustatus = mInstrument.ReadString() ' Status sichern
    mInstrument.WriteString ":DISP:SKEY:STAT 0" & vbLf ' Menü des Gerätes deaktivieren
    Delay 250 'Pause damit das Menü ausgeblendet ist
    mInstrument.WriteString ":HCOP:SDUM:DATA?" & vbLf ' Datei von Gerät über Bus ausgeben und Lokal einlesen"
    mFile.File = mInstrument.ReadString()
    mInstrument.WriteString ":DISP:IMAG " & dispstatus & vbLf  ' Screen wieder in Originalzustand
    mInstrument.WriteString ":DISP:SKEY:STAT " & menustatus & vbLf ' Menü des Gerätes wieder in Originalzustand
    mFile.File = Right(mFile.File, Len(mFile.File) - 8) ' Datenmüll am Anfang der Datei abschneiden
    mFile.Filetype = "png"
    mInstrument.IO.Close
    Exit Function
ups:
ErrorhandlerVisa mInstrument, "Picture_E5071C"
End Function

Private Function Waveform3034A() As String
    On Error GoTo ups
    Dim PointsNum As String
    Dim Error_Check As String
    mInstrument.WriteString ":WAVEFORM:FORMAT ASCII" & vbLf
    mInstrument.WriteString ":WAVEFORM:POINTS?" & vbLf
    PointsNum = mInstrument.ReadString()
    Dim currChannel As Integer
    Dim Target As Range
    Dim timewritten As Boolean
    Set Target = Application.ActiveCell
    For currChannel = 1 To 4
        If ChannelActive3034A(currChannel) Then
            Dim waveformdata As String
            waveformdata = getWaveformData3034A(currChannel)
            WriteToWorkbook3034A (ActiveSheet.Cells(Target.row, Target.Column + currChannel - 1)), waveformdata, currChannel
            If Not timewritten Then
                WriteTimebaseToWorkbook3034A Target, waveformdata
                timewritten = True
            End If
        End If
    Next
    'Error Check
    mInstrument.WriteString ":SYSTEM:ERROR?" & vbLf
    Error_Check = mInstrument.ReadString
    ActiveSheet.Cells(Target.row, Target.Column + 5) = Error_Check
    mInstrument.IO.Close
    mFile.File = "NoFile"
    mFile.Filetype = "NoFile"
    Exit Function
ups:
    ErrorhandlerVisa mInstrument, "Waveform3034A"
End Function

Private Function WriteTimebaseToWorkbook3034A(ByRef Target As Range, waveformdata As String)
    On Error GoTo ups
    Dim TsXInc As String, TsXRef As String, TsXorigin As String
    mInstrument.WriteString ":WAVEFORM:XINCREMENT?" & vbLf
    TsXInc = mInstrument.ReadString()
    mInstrument.WriteString ":WAVEFORM:XREFERENCE?" & vbLf
    TsXRef = mInstrument.ReadString()
    mInstrument.WriteString ":WAVEFORM:XORIGIN?" & vbLf
    TsXorigin = mInstrument.ReadString()
    'ArraySplit = Split(waveformdata, ",")
    Target.Parent.Cells(Target.Cells(1, 1).row, Target.Cells(1, 1).Column) = "Time"
    Dim M As Long
    For M = 0 To UBound(Split(waveformdata, ","))
        Target.Parent.Cells(Target.Cells(1, 1).row + M + 1, Target.Cells(1, 1).Column) = ((M - TsXRef) * TsXInc) + TsXorigin
        'time = [(data point number - xreference) * xincrement] + xorigin
    Next M
    Exit Function
ups:
    ErrorhandlerVisa mInstrument, "WriteTimebaseToWorkbook3034A"
End Function
Private Function WriteToWorkbook3034A(ByRef Target As Range, ByRef waveformdata As String, currChannel As Integer)
    Dim j As Long
    Dim ArraySplit As Variant
    Dim Remove_Header As String
    ArraySplit = Split(waveformdata, ",")
    Target.Parent.Cells(Target.Cells(1, 1).row, Target.Cells(1, 1).Column + 1).Value = "CH" & currChannel
    For j = 1 To UBound(ArraySplit)
        Target.Parent.Cells(Target.Cells(1, 1).row + j + 1, Target.Cells(1, 1).Column + 1).Value = ArraySplit(j)
    Next j
    Remove_Header = ArraySplit(0)
    'The following function splits off the IEEE header from the first data value.
    'This function assumes the header is 10 characters in length, which is
    'consistent for 5000, 6000A/L, and 7000-series scopes.
    Target.Parent.Cells(Target.Cells(1, 1).row + 1, Target.Cells(1, 1).Column + 1).Value = Mid(Remove_Header, 11, (Len(Remove_Header) - 10))
    'Works for newer 2000X, 3000X/T, 4000X, 6000X scopes that have a space following the header.
    'ArraySplit_Header1 = Split(Remove_Header1, " ")
    'Tabelle1.Cells(9, 2) = ArraySplit_Header1(1)
    Exit Function
ups:
    MsgBox "Fehler in Funkion WriteToWorkbook3034A" & vbCrLf & "Fehlermeldung: " & Err.Description & vbCrLf & "Fehler Source: " & Err.source
    End
End Function
